TextInputAction? getInputAction(InputActions action) {
  switch (action) {
    case InputActions.done:
      return TextInputAction.done;
      break;
    case InputActions.next:
      return TextInputAction.next;
      break;
    case InputActions.auto:
    default:
      return null;
      break;
  }
}
